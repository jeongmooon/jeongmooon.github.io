# TABLE SCHEMA

## 테이블 쿼리

```USER_INFO
CREATE TABLE USER_INFO (
	IDX INT(10) NOT NULL AUTO_INCREMENT,
	USER_ID VARCHAR(20),
	PASS VARCHAR(20),
	GAME_CODE VARCHAR(20),
	PRIMARY KEY (idx)
);
```

```PRISONER
CREATE TABLE PRISONER (
	IDX INT(10) NOT NULL AUTO_INCREMENT,
	NAME VARCHAR(20),
	LEVEL VARCHAR(20),
	ICON VARCHAR(20),
	PRIMARY KEY (idx)
);
```

```IDENTITY
CREATE TABLE IDENTITY (
	IDX INT(10) NOT NULL AUTO_INCREMENT,
	PRISONER_IDX INT(10),
	NAME VARCHAR(20),
	GRADE CHAR(1),
	MAIN_IMG CHAR(1),
	PRIMARY KEY (idx)
);
```

```USER_IDENTITY
CREATE TABLE USER_IDENTITY (
	IDX INT(10) NOT NULL AUTO_INCREMENT,
	USER_IDX INT(10),
	IDENTITY_IDX INT(10),
	PRIMARY KEY (idx)
);
```
